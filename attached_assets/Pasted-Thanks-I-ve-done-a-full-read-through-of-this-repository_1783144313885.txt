```java
public synchronized void shutdownManager() {
    try {
        if (isRunning()) {
            forceStop();
            process.waitFor(5, TimeUnit.SECONDS);
        }
    } catch (InterruptedException ignored) {
        Thread.currentThread().interrupt();
    } finally {

        executor.shutdownNow();
    }
}
```
